00085    int i,numfields;

00088 
00089    /* first initalize internal array */
00090    for (i=0;i<12;i++)
00091    {
00092       hist_month[i]=hist_year[i]=hist_fday[i]=hist_lday[i]=0;
00093       hist_hit[i]=hist_files[i]=hist_site[i]=hist_page[i]=hist_visit[i]=0;
00094       hist_xfer[i]=0.0;
00095    }
00096 
00097    hist_fp=fopen(hist_fname,"r");
00098 
00099    if (hist_fp)
00100    {
00101       if (verbose>1) printf("%s %s\n",msg_get_hist,hist_fname);
00102       while ((fgets(buffer,BUFSIZE,hist_fp)) != NULL)
00103       {
00104          i = atoi(buffer) -1;
00105          if (i>11)
00106          {
00107             if (verbose)
00108                fprintf(stderr,"%s (mth=%d)\n",msg_bad_hist,i+1);
00109             continue;
00110          }
00111 
00112          /* month# year# requests files sites xfer firstday lastday */
00113          numfields = sscanf(buffer,"%d %d %lu %lu %lu %lf %d %d %lu %lu",
00114                        &hist_month[i],
00115                        &hist_year[i],
00116                        &hist_hit[i],
00117                        &hist_files[i],
00118                        &hist_site[i],
00119                        &hist_xfer[i],
00120                        &hist_fday[i],
00121                        &hist_lday[i],
00122                        &hist_page[i],
00123                        &hist_visit[i]);
00124 
00125          if (numfields==8)     /* kludge for reading 1.20.xx history files */
00126          {
00127             hist_page[i] = 0;
00128             hist_visit[i] = 0;
00129          }
00130       }
00131       fclose(hist_fp);
00132    }
00133    else if (verbose>1) printf("%s\n",msg_no_hist);
