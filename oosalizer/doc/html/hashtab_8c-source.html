00155 int put_hnode( char     *str,   /* Hostname  */
00156                int       type,  /* obj type  */
00157                u_long    count, /* hit count */
00158                u_long    file,  /* File flag */
00159                double    xfer,  /* xfer size */
00160                u_long   *ctr,   /* counter   */
00161                u_long    visit, /* visits    */
00162                u_long    tstamp,/* timestamp */
00163                char     *lasturl, /* lasturl */
00164                HNODEPTR *htab)  /* ptr>next  */
00165 {
00166    HNODEPTR cptr,nptr;
00167 
00168    /* check if hashed */
00169    if ( (cptr = htab[hash(str)]) == NULL)
00170    {
00171       /* not hashed */
00172       if ( (nptr=new_hnode(str)) != NULL)
00173       {
00174          nptr->flag  = type;
00175          nptr->count = count;
00176          nptr->files = file;
00177          nptr->xfer  = xfer;
00178          nptr->next  = NULL;
00179          htab[hash(str)] = nptr;
00180          if (type!=OBJ_GRP) (*ctr)++;
00181 
00182          if (visit)
00183          {
00184             nptr->visit=(visit-1);
00185             nptr->lasturl=find_url(lasturl);
00186             nptr->tstamp=tstamp;
00187             return 0;
00188          }
00189          else
00190          {
00191             if (ispage(log_rec.url))
00192             {
00193                if (htab==sm_htab) update_entry(log_rec.url);
00194                nptr->lasturl=find_url(log_rec.url);
00195                nptr->tstamp=tstamp;
00196                nptr->visit=1;
00197             }
00198          }
00199       }
00200    }
00201    else
00202    {
00203       /* hashed */
00204       while (cptr != NULL)
00205       {
00206          if (strcmp(cptr->string,str)==0)
00207          {
00208             if ((type==cptr->flag)||((type!=OBJ_GRP)&&(cptr->flag!=OBJ_GRP)))
00209             {
00210                /* found... bump counter */
00211                cptr->count+=count;
00212                cptr->files+=file;
00213                cptr->xfer +=xfer;
00214 
00215                if (ispage(log_rec.url))
00216                {
00217                   if ((tstamp-cptr->tstamp)>=visit_timeout)
00218                   {
00219                      cptr->visit++;
00220                      if (htab==sm_htab)
00221                      {
00222                         update_exit(cptr->lasturl);
00223                         update_entry(log_rec.url);
00224                      }
00225                   }
00226                   cptr->lasturl=find_url(log_rec.url);
00227                   cptr->tstamp=tstamp;
00228                }
00229                return 0;
00230             }
00231          }
00232          cptr = cptr->next;
00233       }
00234       /* not found... */
00235       if ( (nptr = new_hnode(str)) != NULL)
00236       {
00237          nptr->flag  = type;
00238          nptr->count = count;
00239          nptr->files = file;
00240          nptr->xfer  = xfer;
00241          nptr->next  = htab[hash(str)];
00242          htab[hash(str)]=nptr;
00243          if (type!=OBJ_GRP) (*ctr)++;
00244 
00245          if (visit)
00246          {
00247             nptr->visit = (visit-1);
00248             nptr->lasturl=find_url(lasturl);
00249             nptr->tstamp= tstamp;
00250             return 0;
00251          }
00252          else
00253          {
00254             if (ispage(log_rec.url))
00255             {
00256                if (htab==sm_htab) update_entry(log_rec.url);
00257                nptr->lasturl=find_url(log_rec.url);
00258                nptr->tstamp= tstamp;
00259                nptr->visit=1;
00260             }
00261          }
00262       }
00263    }
00264 
00265    if (nptr!=NULL)
00266    {
00267       /* set object type */
00268       if (type==OBJ_GRP) nptr->flag=OBJ_GRP;            /* is it a grouping? */
00269       else
00270       {
00271          /* check if it's a hidden object */
00272          if ((hide_sites)||(isinlist(hidden_sites,nptr->string)!=NULL))
00273            nptr->flag=OBJ_HIDE;
00274       }
00275    }
00276    return nptr==NULL;
00277 }

00811 int put_inode( char     *str,   /* ident str */
00812                int       type,  /* obj type  */
00813                u_long    count, /* hit count */
00814                u_long    file,  /* File flag */
00815                double    xfer,  /* xfer size */
00816                u_long   *ctr,   /* counter   */
00817                u_long    visit, /* visits    */
00818                u_long    tstamp,/* timestamp */
00819                INODEPTR *htab)  /* hashtable */
00820 {
00821    INODEPTR cptr,nptr;
00822 
00823    if ((str[0]=='-') || (str[0]==0)) return 0;  /* skip if no username */
00824 
00825    /* check if hashed */
00826    if ( (cptr = htab[hash(str)]) == NULL)
00827    {
00828       /* not hashed */
00829       if ( (nptr=new_inode(str)) != NULL)
00830       {
00831          nptr->flag  = type;
00832          nptr->count = count;
00833          nptr->files = file;
00834          nptr->xfer  = xfer;
00835          nptr->next  = NULL;
00836          htab[hash(str)] = nptr;
00837          if (type!=OBJ_GRP) (*ctr)++;
00838 
00839          if (visit)
00840          {
00841             nptr->visit=(visit-1);
00842             nptr->tstamp=tstamp;
00843             return 0;
00844          }
00845          else
00846          {
00847             if (ispage(log_rec.url)) nptr->tstamp=tstamp;
00848          }
00849       }
00850    }
00851    else
00852    {
00853       /* hashed */
00854       while (cptr != NULL)
00855       {
00856          if (strcmp(cptr->string,str)==0)
00857          {
00858             if ((type==cptr->flag)||((type!=OBJ_GRP)&&(cptr->flag!=OBJ_GRP)))
00859             {
00860                /* found... bump counter */
00861                cptr->count+=count;
00862                cptr->files+=file;
00863                cptr->xfer +=xfer;
00864 
00865                if (ispage(log_rec.url))
00866                {
00867                   if ((tstamp-cptr->tstamp)>=visit_timeout)
00868                      cptr->visit++;
00869                   cptr->tstamp=tstamp;
00870                }
00871                return 0;
00872             }
00873          }
00874          cptr = cptr->next;
00875       }
00876       /* not found... */
00877       if ( (nptr = new_inode(str)) != NULL)
00878       {
00879          nptr->flag  = type;
00880          nptr->count = count;
00881          nptr->files = file;
00882          nptr->xfer  = xfer;
00883          nptr->next  = htab[hash(str)];
00884          htab[hash(str)]=nptr;
00885          if (type!=OBJ_GRP) (*ctr)++;
00886

00890             nptr->tstamp= tstamp;
00891             return 0;
00892          }
00893          else
00894          {
00895             if (ispage(log_rec.url)) nptr->tstamp= tstamp;
00896          }
00897       }
00898    }
00899 
00900    if (nptr!=NULL)
00901    {
00902       /* set object type */
