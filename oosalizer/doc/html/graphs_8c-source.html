00694 struct pie_data *calc_arc(float min, float max)
00695 {
00696    static struct pie_data data;
00697    double d;
00698 
00699    /* Calculate max line */
00700    d=max;
00701    data.x=cos(d*(2*PI))*((XRAD-2)/2)+CX;
