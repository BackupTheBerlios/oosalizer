00073 /* FMT_LOGREC - terminate log fields w/zeros */
00074 /*********************************************/
00075 
00076 void fmt_logrec(char *buffer)
00077 {
00078    char *cp=buffer;
00079    int  q=0,b=0,p=0;
00080 
00081    while (*cp != '\0')
00082    {
00083       /* break record up, terminate fields with '\0' */
00084       switch (*cp)
00085       {
00086        case ' ': if (b || q || p) break; *cp='\0'; break;
00087        case '"': q^=1;  break;
00088        case '[': if (q) break; b++; break;
00089        case ']': if (q) break; if (b>0) b--; break;
00090        case '(': if (q) break; p++; break;
00091        case ')': if (q) break; if (p>0) p--; break;
00092       }
00093       cp++;
00094    }
00095 }

00134 int parse_record_ftp(char *buffer)
00135 {
00136    int size;
00137    int i,j;
00138    char *cp1, *cp2, *cpx, *cpy, *eob;
00139 
00140    size = strlen(buffer);                 /* get length of buffer        */
00141    eob = buffer+size;                     /* calculate end of buffer     */
00142    fmt_logrec(buffer);                    /* seperate fields with \0's   */
00143 
00144    /* Start out with date/time       */
00145    cp1=buffer;
00146    while (*cp1!=0 && cp1<eob) cp1++;
00147    while (*cp1==0) cp1++;
00148    cpx=cp1;       /* save month name */
00149    while (*cp1!=0 && cp1<eob) cp1++;
00150    while (*cp1==0) cp1++;
00151    i=atoi(cp1);   /* get day number  */
00152    while (*cp1!=0 && cp1<eob) cp1++;
00153    while (*cp1==0) cp1++;
00154    cpy=cp1;       /* get timestamp   */
00155    while (*cp1!=0 && cp1<eob) cp1++;
00156    while (*cp1==0) cp1++;
00157    j=atoi(cp1);   /* get year        */
00158 
00159    /* minimal sanity check */
00160    if (*(cpy+2)!=':' || *(cpy+5)!=':') return 0;
00161    if (j<1990 || j>2100) return 0;
00162    if (i<1 || i>31) return 0;
00163 
00164    /* format date/time field         */
00165    snprintf(log_rec.datetime, sizeof(log_rec.datetime),"[%02d/%s/%4d:%s -0000]",i,cpx,j,cpy);
00166 
00167    /* skip seconds... */
00168    while (*cp1!=0 && cp1<eob) cp1++;
00169    while (*cp1==0) cp1++;
00170    while (*cp1!=0 && cp1<eob) cp1++;
00171    while (*cp1==0) cp1++;
00172 
00173    /* get hostname */
00174    cp2=log_rec.hostname;
00175    strncpy(cp2,cp1,MAXHOST-1);
00176 
00177    /* get filesize */
00178    while (*cp1!=0 && cp1<eob) cp1++;
00179    while (*cp1==0) cp1++;
00180    if (*cp1<'0'||*cp1>'9') log_rec.xfer_size=0;
00181    else log_rec.xfer_size = strtoul(cp1,NULL,10);
00182 
00183    /* URL stuff */
00184    while (*cp1!=0 && cp1<eob) cp1++;
00185    while (*cp1==0) cp1++;
00186    cpx=cp1;
00187    /* get next field for later */
00188    while (*cp1!=0 && cp1<eob) cp1++;
00189    while (*cp1==0) cp1++;
00190    if (strlen(cpx)>MAXURL-20) *(cpx+(MAXURL-20))=0;
00191 
00192    /* skip next two */
00193    while (*cp1!=0 && cp1<eob) cp1++;
00194    while (*cp1==0) cp1++;
00195    while (*cp1!=0 && cp1<eob) cp1++;
00196    while (*cp1==0) cp1++;
00197 
00198    /* fabricate an appropriate request string based on direction */
00199    if (*cp1=='i') snprintf(log_rec.url, sizeof(log_rec.url),"\"POST %s HTTP/1.0\"",cpx);
00200       else        snprintf(log_rec.url, sizeof(log_rec.url),"\"GET %s HTTP/1.0\"",cpx);
00201 
00202    if (cp1<eob) cp1++;
00203    if (cp1<eob) cp1++;
00204    while (*cp1!=0 && cp1<eob) cp1++;
00205    if (cp1<eob) cp1++;
00206    cp2=log_rec.ident;
00207    while (*cp1!=0 && cp1<eob) *cp2++ = *cp1++;
00208    *cp2='\0';
00209 
00210    /* return appropriate response code */
00211    log_rec.resp_code=(*(eob-2)=='i')?206:200;
00212 
00213    return 1;
00214 }
